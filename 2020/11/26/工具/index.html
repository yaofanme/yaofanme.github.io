vsftpd的配置

  • 安装

    yum -y install vfstpd
rpm -qa | grep * # 查看是否安装了某软件

  • 默认配置文件在/etc/vsftpd/vfstpd.conf

  • 创建虚拟用户

    sudo mkdir ftpfile
sudo useradd ftpuser -d /ftpfile -s /sbin/nologin
sudo chown -R ftpuser.ftpuser /ftpfile/
sudo passwd ftpuser
# 配置
cd /etc/vsftpd
# 配置vsftpd
sudo vim /etc/vsftpd/vsftpd.conf
sudo vim chroot_list 新增ftpuser
sudo vim /etc/selinux/config修改为SELINUX=disabled
sudo setenforce 0
# 验证的时候碰到550拒绝访问执行
sudo setsebool -P ftp_home_dir 1
# 设置开机启动
sudo chkconfig vsftpd on

  • 防火墙配置

    sudo vim /etc/sysconfig/iptables
# 配置如下
-A INPUT -p tcp --dport 61001:62000 -j ACCEPT
-A OUTPUT -p tcp --sport 61001:62000 -j ACCEPT
-A INPUT -p tcp --dport 20 -j ACCEPT
-A OUTPUT -p tcp --dport 20 -j ACCEPT
-A INPUT -p tcp --dport 21 -j ACCEPT
-A OUTPUT -p tcp --dport 21 -j ACCEPT
  
# 保存退出，重启防火墙
sudo service iptables restart
# vfstpd验证
sudo service vsftpd restart
# 查看ip，在浏览器进行登录验证
ftp://ip

Nginx的安装配置

简介

  • 可直接支持Rails和PHP的程序
  • 可作为http反向代理服务器
  • 可作为邮件代理服务器
  • 作为负载均衡服务器
  • 帮助实现前后端动静分离

安装

# 安装依赖,nginx默认安装在/usr/local/nginx
sudo yum install gcc
sudo yum install pcre-devel
sudo yum install zlib zlib-devel
# 安装openssl,若需要支持ssl,才需安装openssl
sudo yum install openssl openssl-devel

# Nginx常用命令
sudo ${nginx}/sbin/nginx -t # 测试配置文件
sudo ${nginx}/sbin/nginx # 启动
sudo ${nginx}/sbin/nginx -s stop # 停止
sudo nginx -s quit # 停止
sudo ${nginx}/sbin/nginx -s reload # 重启
sudo ps -ef | grep nginx # 查看进程命令
sudo kill -HUP pid # 平滑重启

# 增加防火墙访问权限
sudo vim /etc/sysconfig/iptables
# 加入配置
-A INPUT -p tcp -m state --state NEW -m tcp --dport 80 -j ACCEPT
# 重启防火墙
sudo service iptables restart

# 配置步骤
sudo vim /usr/local/nginx/conf/nginx.conf
# 在以上文件中添加
include vhost/*.conf

# 在/usr/local/nginx/conf/新建vhost文件夹
/usr/local/nginx/conf/vhost

安装jdk

  • 清理系统默认自带jdk

    # 查看自带的jdk
rpm -qa | grep jdk
# 之后进行卸载
sudo yun remove xxx
# 修改权限
sudo a+x jdk7.rpm
# 安装,默认路径/usr/local
sudo rpm -ivh jdk7.rpm
# 配置环境变量
sudo vim /etc/profile
# 添加以下内容
export JAVA_HOME=/usr/java/jdk1.7.0_80
export CLASSPATH=.:$JAVA_HOME/jre/lib/rt.jar:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar
export CATALINA_HOME=/home/yaofan/tomcat
export PATH=$JAVA_HOME/bin:$PATH:$CATALINA_HOME/bin

安装tomcat

  • 下载之后配置环境变量

    sudo vim /etc/profile
# 在最下方添加
export CATALINA_HOME=tomcat path
# 配置UTF-8字符集
cd ${CATALINA_HOME}/conf
sudo vim server.xml
# 找到配置8080默认端口的位置，在xml节点末尾增加URIEncoding="UTF-8"

安装Mysql

  • 安装

    sudo yum -y install mysql mysql-server mysql-devel
sudo service mysqld start
# 进入mysql
mysql -u root
# 修改密码
use mysql;
update user set password=password('root') where user='root';
# 开放远程登录权限
GRANT ALL PRIVILEGES ON * . * TO 'root'@'%' IDENTIFIED BY 'root' WITH GRANT OPTION;
FLUSH PRIVILEGES;
# 默认配置文件在/etc/my.cnf
# 配置字符集
sudo vim /etc/my.cnf
# 添加配置,在mysqld节点下添加
default-character-set=utf8
character-set-server=utf8
# 自启动配置
sudo chkconfig mysqld on
# 防火墙配置
sudo vim /etc/sysconfig/iptables
# 添加配置,开放3306端口
-A INPUT -p tcp -m tcp --dport 3306 -j ACCEPT

Git的配置

  • Git初始化配置

    git config --global user.name "yaofan"
git config --global user.email "yaofan97@gmail"

  • 设置SSH Key

    ssh-keygen -t rsa -C "yaofan97@gmail"

  • 查看公钥

    cat ~/.ssh/id_rsa.pub

  • 与github通信

    ssh -T git@github.com

  • 常见Git操作

    # git初始化
git init
# 查看状态
git status
# 添加文件
git add .
# 本地提交
git commit -am 'first commit init project'
# 添加远程仓库
git remote add origin + URL
# 将本地推送到远程仓库,-f强制推送
git push -u origin master、
# 查看远程分支
git branch -r
# 创建分支
git checkout -b v1.0 origin/master
# 推送分支到远程
git push origin HEAD -u
